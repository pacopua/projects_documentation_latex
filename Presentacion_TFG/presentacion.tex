\begin{frame}{El problema a resolver}
    \vspace{-0.8cm}
    \begin{columns}[T]
        \begin{column}{0.38\textwidth}
            \vspace{0.3cm}
            El uso impulsivo de asistentes de IA genera un \textbf{ciclo vicioso} que degrada la calidad del código:
            \vspace{0.5cm}
            \begin{itemize}
                \item Prompt impulsivo
                \item Código sin estructura
                \item Bugs

                \item Deuda técnica

            \vspace{0.3cm}

            \textbf{No funcionales:}
            \vspace{0.15cm}
            \begin{itemize}
                \footnotesize
                \item \textbf{Trazabilidad:} entre especificaciones, tareas y código generado.
                \item \textbf{Integración:} con al menos dos asistentes distintos para evitar vendor lock-in.
                \item \textbf{Modularidad:} añadir módulos sin alterar los existentes.
            \end{itemize}
        \end{column}
    \end{columns}

\begin{comment}
    \begin{frame}{La diferencia: \textit{colbPowers}}
        \vspace{-0.7cm}
        \textit{superpowers} ofrece la arquitectura base (orquestador + subagentes vía skills). \textbf{colbPowers} añade:
    
        \begin{columns}[T]
            \begin{column}{0.48\textwidth}
                \begin{block}{Contexto persistente entre sesiones}
                    \footnotesize
                    \texttt{constitution.md}: visión del proyecto, stack tecnológico y estándares no negociables.\\[0.2cm]
                    \texttt{features.md}: roadmap con criterios de aceptación por funcionalidad.
                \end{block}
                \vspace{0.2cm}
                \begin{block}{Configuración interactiva}
                    \footnotesize
                    Skills \textit{defining-constitution} y \textit{defining-features}: el agente construye estos documentos junto al desarrollador en la primera sesión.
                \end{block}
            \end{column}
            \begin{column}{0.48\textwidth}
                \begin{block}{Flujo SDD}
                    \footnotesize
                    Skill \textit{using-colbPowers} (adaptación de \textit{using-superpowers}) inyectada vía hook \texttt{SessionStart}: las reglas SDD activas \textbf{desde el primer mensaje}.
                \end{block}
                \vspace{0.2cm}
                \begin{block}{Subagentes aislados}
                    \footnotesize
                    Mecanismo de \textit{superpowers}; arrancan sin acceso al historial de conversación, solo ven el contexto que el orquestador les proporciona explícitamente..
                \end{block}
            \end{column}
        \end{columns}
        \vspace{0.1cm}
    \end{frame}
\end{comment}

\begin{frame}{Sostenibilidad}

            \centering\textbf{Económica}
            \rule{\linewidth}{0.4pt}\\[0.2cm]

        \end{column}
        \begin{column}{0.32\textwidth}
            \centering\textbf{Ambiental}
            \rule{\linewidth}{0.4pt}\\[0.2cm]
            \footnotesize
            Consumo inicial elevado (sobrecoste estructural \textbf{$\times$2,2--$\times$4,5} respecto al baseline), pero genera documentación que reduce el uso posterior de IA.\\[0.3cm]
            Los tests de prueba reducen la cantidad de \textit{prompts} para solucionar errores inesperados.
        \end{column}
        \begin{column}{0.32\textwidth}
            \centering\textbf{Social}
            \rule{\linewidth}{0.4pt}\\[0.2cm]
            \footnotesize
            La metodología Human-in-the-loop conlleva mayor involucra\-miento del desarrollador como profesional.\\[0.3cm]
            La generación de documentación facilita la incorporación de talento nuevo en los proyectos.
        \end{column}

% SLIDE 21: METODOLOGÍA – MÉTRICAS DE CALIDAD
% ============================================================
\begin{frame}{Metodología -- Métricas de calidad}
    \vspace{-0.5cm}
    \textbf{Complejidad ciclomática} {\footnotesize\textcolor{gray}{(McCabe, 1976)}}:
    \begin{equation*}
        \mathcal{CC}(G) = |E| - |V| + 2p
    \end{equation*}
    \vspace{-0.35cm}
    {\footnotesize\color{gray}
    \begin{itemize}\setlength{\itemsep}{1pt}
        \item $G$: grafo de flujo de control de la función (nodos\,=\,bloques básicos, aristas\,=\,saltos de control)
        \item $p$: Componentes conexas del grafo.
    \end{itemize}}

    \vspace{0.1cm}
    \textbf{Índice de mantenibilidad} {\footnotesize\textcolor{gray}{(Halstead, 1977; Oman \& Hagemeister, 1992)}}:
    \begin{equation*}
        V_H = (N_1 + N_2)\log_2(\eta_1 + \eta_2)
    \end{equation*}
    \vspace{-0.35cm}
    {\footnotesize\color{gray}
    \begin{itemize}\setlength{\itemsep}{1pt}
        \item $N_1, N_2$: ocurrencias \emph{totales} de operadores y operandos en el código
        \item $\eta_1, \eta_2$: número de operadores y operandos \emph{distintos}
    \end{itemize}}
    \vspace{-0.4cm}
    \begin{equation*}

\begin{frame}{Conclusiones -- Resultados experimentales}
    \textbf{Resultados mixtos:}
    \vspace{0.1cm}
    \begin{itemize}
        \item \textbf{Caso 2 (proyecto nuevo) con OpenCode}: $-$27\% complejidad ciclomática, $+$15\% índice de mantenibilidad ($p < 0{,}05$, efecto grande).
        \vspace{0.05cm}
        \item \textbf{Caso 2 con ClaudeCode}: variaciones no significativas ($p > 0{,}5$); ClaudeCode ya produce código de mayor calidad en baseline.
        \vspace{0.05cm}
        \item \textbf{Caso 1 (repo existente)}: patrón invertido --- mayor complejidad ciclomática con colbPowers; sin mejora neta en mantenibilidad. Los agentes apenas invocaron \textit{code-review-graph}.
        \vspace{0.05cm}
        \item \textbf{Sobrecoste de tokens:} $\times$2,2 -- $\times$4,5 respecto al baseline, constante e independiente del resultado.
    \end{itemize}
    \vspace{0.1cm}
    \footnotesize En producción en \textit{Colb.AI} la valoración ha sido \textbf{positiva}: los modelos de producción son más capaces que los experimentales y los documentos de contexto reales tienen mayor calidad y exhaustividad.
